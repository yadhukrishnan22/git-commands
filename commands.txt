git config --global user.name "username" [to create a user in git]

git config --global user.email "email" [to create a mail for user]

git init [to initialize git in a new folder. should be only done once]

git status [to check untracked files]

git add -A [to all untracked files to stagged]

git commit -m "commit message" [to add all stagged files]

git remote add remote_name remote_path []

git push remote_name branch_name [to push code ]

git branch [to list all branches]

git version -v [to find the remote name]
